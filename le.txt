class Solution(object):
    def longestConsecutive(self, nums):
        nums.sort()
        b=0
        c=1
        if(len(nums)>0):
            for i in range(len(nums)-1):
                if(nums[i+1]-nums[i]==1):
                    c=c+1
                elif(nums[i+1]==nums[i]):
                    c=c+0
                else:
                    b=max(b,c)
                    c=1
            b=max(b,c)
            return b
        else:
            return 0
